Non-Local Backup Redundancy

This query finds SQL databases whose current and requested backup storage redundancy are not set to Local, parameterizes your subscription list, and maps IDs to friendly names without exposing GUIDs.

// parameterize your subscriptions instead of embedding GUIDs
let SubscriptionList = dynamic([
    "",
    ""
    // …add more as needed
]);

// map those IDs to friendly names in a separate table
let SubscriptionMap = datatable (subscriptionId:string, subscriptionName:string) [
    "", "",
    "", ""
    // …etc.
];

resources
| where type =~ "Microsoft.Sql/servers/databases"
| extend 
    currentBackupStorageRedundancy   = tostring(properties.currentBackupStorageRedundancy),
    requestedBackupStorageRedundancy = tostring(properties.requestedBackupStorageRedundancy)
| where currentBackupStorageRedundancy  != "Local"
| where requestedBackupStorageRedundancy != "Local"
| where name != "master"
| where subscriptionId in (SubscriptionList)
// join in friendly subscription names
| join kind=leftouter SubscriptionMap on subscriptionId
| extend SubscriptionName = coalesce(subscriptionName, "Unknown")
// project only the non-sensitive fields you need
| project
    Name                 = name,
    Location             = location,
    ResourceGroup        = resourceGroup,
    SubscriptionName,
    Application          = tostring(tags.Application),
    Department           = tostring(tags.Department),
    StorageRedundancy    = currentBackupStorageRedundancy
