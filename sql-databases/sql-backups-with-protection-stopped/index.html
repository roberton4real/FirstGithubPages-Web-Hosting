Stopped SQL Backups

This query surfaces SQL database backups whose protection state is ProtectionStopped, parameterizes your subscription list, and maps IDs to friendly names without exposing any raw GUIDs.

// parameterize your subscriptions instead of embedding GUIDs
let SubscriptionList = dynamic([
    "",
    ""
    // …add more as needed
]);

// map those IDs to friendly names in a separate table
let SubscriptionMap = datatable (subscriptionId:string, subscriptionName:string) [
    "", "",
    "", ""
    // …etc.
];

recoveryservicesresources
| where type in~ (
    "Microsoft.RecoveryServices/vaults/backupFabrics/protectionContainers/protectedItems"
)

// capture policy ID
| extend policyId = properties.policyInfo.policyId

// extract vault name from the resource ID
| extend vaultName = split(split(id, "/Microsoft.RecoveryServices/vaults/")[1], "/")[0]

// derive vault resource path prefix
| extend vaultId = tolower(split(id, "/backupFabrics/")[0])

// combine management and workload type
| extend dataSourceType = strcat(properties.backupManagementType, "/", properties.workloadType)

// friendly display name
| extend friendlyName = properties.friendlyName

// source resource's RG and subscription
| extend
    dsResourceGroup = split(split(properties.dataSourceInfo.resourceID, "/resourceGroups/")[1], "/")[0],
    dsSubscription  = split(split(properties.dataSourceInfo.resourceID, "/subscriptions/")[1], "/")[0]

// the actual protected resource path
| extend resourceId = tolower(tostring(properties.dataSourceSetInfo.resourceID))

// determine last restore point
| extend lastRestorePoint = case(
    isnull(properties.lastRecoveryPoint), properties.lastBackupTime,
    properties.lastRecoveryPoint
)

// metadata
| extend
    primaryLocation   = properties.dataSourceInfo.resourceLocation,
    parentType        = properties.parentType,
    instanceName      = case(
        isnotnull(properties.serverName),
        strcat(properties.serverName, "/", properties.parentName),
        "NA"
    ),
    policyName        = case(
        extract(@"([^/]*)/backupPolicies/([^/]*)", 2, tostring(policyId)) != "",
        extract(@"([^/]*)/backupPolicies/([^/]*)", 2, tostring(policyId)),
        properties.policyInfo.name
    ),
    protectionState   = properties.currentProtectionState,
    managedIdentity   = iif(
        properties.identityDetails.useSystemAssignedIdentity == false,
        "userassigned",
        "systemassigned"
    )

// filter to only stopped protections on SQL databases
| where protectionState in~ ("ProtectionStopped")
| where dataSourceType in~ ("AzureWorkload/SQLDataBase")

// only include the subscriptions you care about
| where subscriptionId in (SubscriptionList)

// join in friendly subscription names
| join kind=leftouter SubscriptionMap on subscriptionId
| extend SubscriptionName = coalesce(subscriptionName, "Unknown")

// select only the non-sensitive fields you need
| project
    Name             = friendlyName,
    Location         = primaryLocation,
    ResourceGroup    = resourceGroup,
    SubscriptionName,
    LastRestorePoint = lastRestorePoint,
    ProtectionState
| order by Name asc
