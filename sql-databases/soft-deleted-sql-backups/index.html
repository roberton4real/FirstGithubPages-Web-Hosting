Soft-Deleted SQL Backups

This query surfaces SQL database backups that are currently in the SoftDeleted state, parameterizes your subscription list, and maps IDs to friendly names without exposing any raw GUIDs.

// parameterize your subscriptions instead of embedding GUIDs
let SubscriptionList = dynamic([
    "",
    ""
    // …add more as needed
]);

// map those IDs to friendly names in a separate table
let SubscriptionMap = datatable (subscriptionId:string, subscriptionName:string) [
    "", "",
    "", ""
    // …etc.
];

recoveryservicesresources
| where type =~ "Microsoft.RecoveryServices/vaults/backupFabrics/protectionContainers/protectedItems"

// extract the vault name dynamically from the resource ID
| extend vaultName = split(split(id, "/Microsoft.RecoveryServices/vaults/")[1], "/")[0]

// get the data-source type
| extend dataSourceType = properties.backupManagementType

// flag soft-deleted items and their retention period
| extend
    isSoftDeleted           = tostring(properties.protectionState),
    softDeleteRetentionDays = toint(properties.softDeleteRetentionPeriod)

// filter to only SQLDatabase backups that are currently soft-deleted
| where isSoftDeleted == "SoftDeleted"
| where dataSourceType in~ ("AzureWorkload/SQLDatabase")

// only include the subscriptions you care about
| where subscriptionId in (SubscriptionList)

// join in friendly subscription names
| join kind=leftouter SubscriptionMap on subscriptionId 
| extend SubscriptionName = coalesce(subscriptionName, "Unknown")

// select only the non-sensitive fields you need
| project
    Name                       = name,
    Location                   = location,
    ResourceGroup              = resourceGroup,
    SubscriptionName,
    VaultName,
    IsSoftDeleted              = isSoftDeleted,
    SoftDeleteRetentionDays
| order by Name asc
