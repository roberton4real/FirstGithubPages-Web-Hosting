Failed RSV backups for SQL databases

This query surfaces SQL databases whose Recovery Services Vault backups have:

  • Protection status Unhealthy
  • Protection state Protected
  • Last backup status Failed
  • A non-empty backup time within the past 7 days
// Parameterized subscription list (or remove to query all)
let subscriptionList = dynamic([
    "",
    ""
]);

recoveryservicesresources
| where type == "microsoft.recoveryservices/vaults/backupfabrics/protectioncontainers/protecteditems"
| where properties.protectionStatus == "Unhealthy"
| where properties.protectionState   == "Protected"
| where properties.lastBackupStatus  == "Failed"
| where properties.workloadType in~ ("SQLDataBase")
| where isnotempty(properties.lastBackupTime)
      and todatetime(properties.lastBackupTime) >= ago(7d)
| where subscriptionId in (subscriptionList)      // remove to include all
| extend ResourceLink = strcat(
      'https://portal.azure.com/#@', tenantId,
      '/resource', id
    )
| project
    Name                 = tostring(properties.friendlyName),
    Location             = location,
    ResourceGroup        = resourceGroup,
    SubscriptionId       = subscriptionId,
    WorkloadType         = tostring(properties.workloadType),
    LastBackupStatus     = tostring(properties.lastBackupStatus),
    ProtectionStatus     = tostring(properties.protectionStatus),
    LastBackupTime       = todatetime(properties.lastBackupTime),
    BackupManagementType = tostring(properties.backupManagementType),
    VaultId              = tostring(properties.vaultId),
    ResourceLink
| order by LastBackupTime desc
